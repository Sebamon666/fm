tabla = df_2025[df_2025["Tiempo"].dt.dayofweek < 5].copy()
tabla["Da"] = tabla["Tiempo"].dt.dayofweek
tabla["Hora"] = tabla["Tiempo"].dt.hour

heatmap_data = tabla.groupby(["Da", "Hora"]).size().unstack(fill_value=0)
dias = ["Lun", "Mar", "Mi", "Jue", "Vie"]
heatmap_data.index = dias

colores = ["#d3d3d3", "#ff9999", "#ff0000"]  # gris claro  rojo claro  rojo intenso
cmap_custom = LinearSegmentedColormap.from_list("gris_a_rojo", colores)

plt.figure(figsize=(12, 6))
sns.heatmap(heatmap_data, cmap=cmap_custom, linewidths=0.5, linecolor="gray", cbar_kws={"label": "Cantidad de accesos"})
plt.title("Heatmap de accesos por da y hora")
plt.xlabel("Hora del da")
plt.ylabel("Da de la semana")
plt.tight_layout()
plt.show()
